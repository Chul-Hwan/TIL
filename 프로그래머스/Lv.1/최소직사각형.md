[문제 설명]
여러가지의 명함들이 있는데, 그 명함이 다 수납할 수 있는 지갑을 만드는 것이 목표.

## 내 풀이
``` C#
using System;

public class Solution {
    public int solution(int[,] sizes) {
        int answer = 0;
        int w = 0;
        int h = 0;
    
        for(int i = 0; i < sizes.GetLength(0); i++)
        {
            int tempW = sizes[i,0] ; 
            int tempH = sizes[i,1];
            
            if(tempW < tempH){
                sizes[i,0] = tempH;
                sizes[i,1] = tempW;
            }
                
            if(sizes[i,0] > w) w = sizes[i,0];
            if(sizes[i,1] > h) h = sizes[i,1];
        }
        
        return w*h;
    }
}
```
